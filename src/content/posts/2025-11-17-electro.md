$$
\nabla \cdot (\frac{\vec{R}}{R^3}) = 4\pi \delta(\vec{R})
$$

$$
\nabla (\vec{p} \cdot \vec{r}) = \vec{p}
$$

$$
\varphi = \frac{1}{4\pi\epsilon_0} \frac{\vec{p} \cdot \vec{r}}{r^3}
$$

$$
\vec{E} = -\frac{1}{4\pi\epsilon_0} \nabla (\frac{\vec{p} \cdot \vec{r}}{r^3})
$$

$$
= -\frac{1}{4\pi\epsilon_0} (\frac{\vec{p}}{r^3} + (\vec{p} \cdot \vec{r}) (-3\frac{\hat{r}}{r^4}))
$$

$$
= -\frac{1}{4\pi\epsilon_0} \frac{\vec{p} - 3(\vec{p} \cdot \hat{r})\hat{r}}{r^3}
$$

$$
d\vec{F}_{12} = \frac{\mu_0}{4\pi} \frac{\vec{j}_1 d\tau_1 \times (\vec{j}_2 d\tau_2 \times \hat{r}_{12})}{r_{12}^2} \quad d\vec{F}_{12} \neq d\vec{F}_{21}
$$

$$
\vec{B} = \frac{\mu_0}{4\pi} \int \frac{\vec{j}d\tau' \times \vec{r}}{r^2} \quad \frac{\vec{r}}{r^2} = -\nabla(\frac{1}{r})
$$

$$
\vec{B} = \frac{\mu_0}{4\pi} \int \nabla(\frac{1}{r}) \times \vec{j}(\vec{r}') d\tau'
$$

$$
= \frac{\mu_0}{4\pi} \int \left[ \nabla \times (\frac{\vec{j}(\vec{r}')}{r}) - \frac{\nabla \times \vec{j}(\vec{r}')}{r} \right] d\tau'
$$

$$
= \nabla \times \left( \frac{\mu_0}{4\pi} \int \frac{\vec{j}(\vec{r}')}{r} d\tau' \right) \quad \vec{A}(\vec{r})
$$

$$
\vec{a} \times (\vec{b} \times \vec{c}) = \vec{b}(\vec{a} \cdot \vec{c}) - \vec{c}(\vec{a} \cdot \vec{b})
$$

$$
\nabla \times \vec{B} = \nabla \times (\nabla \times \vec{A}) = \nabla(\nabla \cdot \vec{A}) - \nabla^2 \vec{A}
$$

$$
\nabla \cdot \vec{A} = \frac{\mu_0}{4\pi} \nabla \cdot \int \frac{\vec{j}(\vec{r}')}{R} d\tau' = \frac{\mu_0}{4\pi} \int \vec{j}(\vec{r}') \cdot \nabla(\frac{1}{R}) d\tau'
$$

$$
= -\frac{\mu_0}{4\pi} \int \vec{j}(\vec{r}') \cdot \nabla'(\frac{1}{R}) d\tau' = -\frac{\mu_0}{4\pi} \int \left( \nabla' \cdot (\frac{\vec{j}(\vec{r}')}{R}) - \frac{1}{R} \underbrace{\nabla' \cdot \vec{j}(\vec{r}')}_{0} \right) d\tau'
$$

$$
-\nabla^2 \vec{A} = -\frac{\mu_0}{4\pi} \nabla^2 \int \frac{\vec{j}(\vec{r}') d\tau'}{R} = \int \frac{\mu_0}{4\pi} (-4\pi \delta(\vec{r}-\vec{r}')) \vec{j}(\vec{r}') d\tau'
$$

$$
= \mu_0 \vec{j}(\vec{r})
$$

$$
\Rightarrow \nabla \times \vec{B} = \mu_0 \vec{j}(\vec{r})
$$

对于磁偶极子

$$
\vec{A} = \frac{\mu_0 I}{4\pi} \oint \frac{d\vec{l}}{R} = \frac{\mu_0 I}{4\pi} \oint \frac{d\vec{l}}{|\vec{r}-\vec{r}'|} \approx \frac{\mu_0 I}{4\pi r^3} \int (\vec{r} \cdot \vec{r}') d\vec{l}'
$$

$$
= \frac{\mu_0 I}{4\pi r^3} \cdot \frac{1}{2} \oint (\vec{r}' \times d\vec{l}') \times \vec{r}
$$

$$
\vec{B} = \frac{\mu_0}{4\pi} \nabla \times (\frac{\vec{m} \times \vec{r}}{r^3}) = \frac{\mu_0}{4\pi} \left( (\nabla \cdot \frac{\vec{r}}{r^3})\vec{m} - (\vec{m} \cdot \nabla)\frac{\vec{r}}{r^3} \right)
$$

$$
= -\frac{\mu_0}{4\pi} (\vec{m} \cdot \nabla) \frac{\vec{r}}{r^3} = -\frac{\mu_0}{4\pi} \frac{\vec{m} - 3(\vec{m} \cdot \hat{r})\hat{r}}{r^3}
$$
